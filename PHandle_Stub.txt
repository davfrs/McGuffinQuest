bool playerAction (char action) {
    bool acted = false;
    if (action == '5') {
        acted = true;
    }
    else if (action == '6') {
        acted = game.player.moveNorth();
    }
    else if (action == '4') {
        acted = game.player.moveSouth();
    }
    else if (action == '8') {
        acted = game.player.moveWest();
    }
    else if (action == '2') {
        acted = game.player.moveEast();
    }
    else if (action == '>') {
        acted = game.player.moveDown();
    }
    else if (action == '<') {
        acted = game.player.moveUp();
    }
    return acted;
}
